package Com.fss.pms;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class DeleteEmp {
	
	public static String DRIVER_CLASS_NAME = "com.mysql.cj.jdbc.Driver";
	public static String CONNECTION_URL = "jdbc:mysql://localhost:3306";
	public static String CONNECTION_USER = "root";
	public static String CONNECTION_PASSWORD = "Saloni@14";
	
	public static void main(String[] args) {
		
		String deleteSQL = "DELETE FROM pms.employee WHERE emp_ID = ?";
		 
		try (Connection connection = getConnection();
				PreparedStatement preparedStatement = connection
						.prepareStatement(deleteSQL)) {
            //giving empid=1
			preparedStatement.setInt(1, 1);
 
			int rowCount = preparedStatement.executeUpdate();
 
			System.out
					.println("Record Deleted successfully from database. Row Count returned is :: "
							+ rowCount);
 
		} catch (SQLException e) {
			System.out
					.println("An exception occured while deleting data from database. Exception is :: "
							+ e);
		}
 
	}
 
	public static Connection getConnection() {
 
		try {
			Class.forName(DRIVER_CLASS_NAME);
		} catch (ClassNotFoundException e) {
			System.out.println("Error while registering JDBC driver");
			return null;
		}
 
		Connection connection = null;
		try {
			connection = DriverManager.getConnection(CONNECTION_URL,
					CONNECTION_USER, CONNECTION_PASSWORD);
 
		} catch (SQLException e) {
			System.out.println("Failed to create Connection");
			return null;
		}
		System.out.println("Connection created Successfully.");
		return connection;
	}
}

create database PMS;

create table payroll(
payroll_ID int primary key,

job_ID int,
salary_ID varchar(45),
leave_ID varchar(45),
date date,
report varchar(200),
total_amount double,
foreign key(emp_ID) references employee(emp_ID),

foreign key(salary_ID) references salary_Bonus(salary_ID),

create table job_Department(
job_ID int primary key,
job_dept varchar(50),

leave_ID varchar(45) primary key,
